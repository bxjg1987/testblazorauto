using System;
using System.Collections.Generic;
using System.Linq;
using System.Security.Cryptography;
using System.Text;

namespace BXJG.Common
{
    public static class RandomHelper
    {
        private static readonly char[] LowercaseChars = "abcdefghijklmnopqrstuvwxyz".ToCharArray();
        private static readonly char[] UppercaseChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".ToCharArray();
        private static readonly char[] NumberChars = "0123456789".ToCharArray();
        private static readonly char[] SafeSpecialChars = "!@#$%^&*()-_=+[]{}|;:,.<>?~".ToCharArray(); // 移除可能引起问题的字符

        public static string GetRandomString(
            int length = 8,
            bool includeLowercase = true,
            bool includeUppercase = true,
            bool includeNumber = true,
            bool includeSpecial = false)
        {
            // 参数验证
            if (length <= 0)
                throw new ArgumentException("Length must be greater than zero.", nameof(length));

            // 收集启用的字符集
            var charSets = new List<char[]>();
            if (includeLowercase) charSets.Add(LowercaseChars);
            if (includeUppercase) charSets.Add(UppercaseChars);
            if (includeNumber) charSets.Add(NumberChars);
            if (includeSpecial) charSets.Add(SafeSpecialChars);

            if (charSets.Count == 0)
                throw new ArgumentException("At least one character set must be included.");

            // 检查长度是否满足最小要求
            if (length < charSets.Count)
                throw new ArgumentException($"Length must be at least {charSets.Count} when including {charSets.Count} character sets.");

            // 创建字符池（所有可用字符）
            var charPool = charSets.SelectMany(chars => chars).ToArray();
            
            // 使用加密级随机数生成器
            using var rng = RandomNumberGenerator.Create();
            
            // 步骤1：确保每个字符集至少包含一个字符
            var result = new StringBuilder(length);
            foreach (var charSet in charSets)
            {
                result.Append(GetRandomCharFromSet(charSet, rng));
            }

            // 步骤2：填充剩余字符
            for (int i = charSets.Count; i < length; i++)
            {
                result.Append(GetRandomCharFromSet(charPool, rng));
            }

            // 步骤3：随机打乱结果字符串
            return ShuffleString(result.ToString(), rng);
        }

        private static char GetRandomCharFromSet(char[] charSet, RandomNumberGenerator rng)
        {
            var randomBytes = new byte[4];
            rng.GetBytes(randomBytes);
            uint randomValue = BitConverter.ToUInt32(randomBytes, 0);
            return charSet[randomValue % (uint)charSet.Length];
        }

        private static string ShuffleString(string input, RandomNumberGenerator rng)
        {
            var chars = input.ToCharArray();
            int n = chars.Length;
            
            // 一次性获取所有需要的随机字节
            var randomBytes = new byte[n * 4];
            rng.GetBytes(randomBytes);

            for (int i = n - 1; i > 0; i--)
            {
                uint randomValue = BitConverter.ToUInt32(randomBytes, i * 4);
                int j = (int)(randomValue % (uint)(i + 1));
                (chars[i], chars[j]) = (chars[j], chars[i]);
            }

            return new string(chars);
        }
    }
}
